\begin{lstlisting}
module Zoo

sig Species {
	space: Int
}

sig Animal {
    cage: one Cage,
    species: one Species
}

sig Cage {
    capacity: Int,
    animals: set Animal
}{
    //#animals <= capacity
	(#animals).mul[animals.species.space] <= capacity
    //all a1, a2: animals | a1 != a2 implies a1.species = a2.species
	one animals.species
}

// Synchronizing Animal.cage and Cage.animals
fact {
    all a: Animal | a in a.cage.animals
    all c: Cage | all a: c.animals | a.cage = c
}

// Defining specific species
one sig Lion extends Species {}{space=1000}
one sig Gnou extends Species {}{space=3000}

// Defining specific animals
one sig Lea, Leo, Lou extends Animal {}{ species = Lion }
one sig gNathan, gNatalie extends Animal {}{ species = Gnou }

// Defining specific cages with different capacities
one sig C1 extends Cage {}{ capacity = 6000 }
one sig C2 extends Cage {}{ capacity = 3000 }

// Ensuring a valid instance exists
run {
    #Animal = 5
    #Cage = 2
} for 14 int
